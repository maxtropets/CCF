85 static KVMap::Handle* _get_map_handle(
86 js::core::Context& jsctx, JSValueConst _this_val)
87 {
88 auto this_val = jsctx.duplicate_value(_this_val);
89 auto map_name_val = this_val["_map_name"];
90 auto map_name = jsctx.to_str(map_name_val);
91
92 if (!map_name.has_value())
93 {
94 LOG_FAIL_FMT("No map name stored on handle");
95 return nullptr;
96 }
97
98 auto& handles = jsctx.globals.kv_handles;
99 auto it = handles.find(map_name.value());
100 if (it == handles.end())
101 {
102 it = handles.emplace_hint(it, map_name.value(), nullptr);
103 }
104
105 if (it->second == nullptr)
106 {
107 kv::Tx* tx = jsctx.globals.tx;
108 if (tx == nullptr)
109 {
110 LOG_FAIL_FMT("Can't rehydrate MapHandle - no transaction context");
111 return nullptr;
112 }
113 it->second = tx->rw<KVMap>(map_name.value());
114 }
115
116 return it->second;
117 }

379 template <HandleGetter handle_getter_>
380 static JSValue js_kv_map_foreach(
381 JSContext* ctx, JSValueConst this_val, int argc, JSValueConst* argv)
382 {
383 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
384
385 if (argc != 1)
386 return JS_ThrowTypeError(
387 ctx, "Passed %d arguments, but expected 1", argc);
388
389 js::core::JSWrappedValue func(ctx, argv[0]);
390 js::core::JSWrappedValue obj(ctx, this_val);
391
392 if (!JS_IsFunction(ctx, func.val))
393 {
394 return JS_ThrowTypeError(ctx, "Argument must be a function");
395 }
396
397 auto handle = handle_getter_(jsctx, this_val);
398 JS_CHECK_HANDLE(handle);
399
400 bool failed = false;
401 handle->foreach(
402 [&jsctx, &obj, &func, &failed](const auto& k, const auto& v) {
403 auto value = jsctx.new_array_buffer_copy(v.data(), v.size());
404 if (value.is_exception())
405 {
406 failed = true;
407 return false;
408 }
409 auto key = jsctx.new_array_buffer_copy(k.data(), k.size());
410 if (key.is_exception())
411 {
412 failed = true;
413 return false;
414 }
415 // JS forEach expects (v, k, map) rather than (k, v)
416 std::vector<js::core::JSWrappedValue> args = {value, key, obj};
417
418 auto val = jsctx.inner_call(func, args);
419
420 if (val.is_exception())
421 {
422 failed = true;
423 return false;
424 }
425
426 return true;
427 });
428
429 if (failed)
430 {
431 return ccf::js::core::constants::Exception;
432 }
433
434 return ccf::js::core::constants::Undefined;
435 }

437 template <HandleGetter HG>
438 static JSValue _create_kv_map_handle(
439 js::core::Context& ctx,
440 const std::string& map_name,
441 MapAccessPermissions access_permission)
442 {
443 // This follows the interface of Map:
444 // https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Map
445 // Keys and values are ArrayBuffers. Keys are matched based on their
446 // contents.
447 auto view_val = ctx.new_obj_class(kv_map_handle_class_id);
448 JS_CHECK_EXC(view_val);
449
450 // Store (owning) copy of map_name in a property on this JSValue
451 auto map_name_val = ctx.new_string(map_name);
452 JS_CHECK_EXC(map_name_val);
453 JS_CHECK_SET(view_val.set("_map_name", std::move(map_name_val)));
454
455 // Add methods to handle object. Note that this is done once, when this
456 // object is created, because jsctx.access is constant. If the access
457 // restrictions could vary between invocations, then this object's
458 // properties would need to be updated as well.
459
460 auto has_fn = js_kv_map_has<HG>;
461 auto get_fn = js_kv_map_get<HG>;
462 auto size_fn = js_kv_map_size_getter<HG>;
463 auto set_fn = js_kv_map_set;
464 auto delete_fn = js_kv_map_delete;
465 auto clear_fn = js_kv_map_clear;
466 auto foreach_fn = js_kv_map_foreach<HG>;
467 auto get_version_fn = js_kv_get_version_of_previous_write<HG>;
468
469 if (access_permission == MapAccessPermissions::ILLEGAL)
470 {
471 has_fn = js_kv_map_has_denied;
472 get_fn = js_kv_map_get_denied;
473 size_fn = js_kv_map_size_denied;
474 set_fn = js_kv_map_set_denied;
475 delete_fn = js_kv_map_delete_denied;
476 clear_fn = js_kv_map_clear_denied;
477 foreach_fn = js_kv_map_foreach_denied;
478 get_version_fn = js_kv_map_get_version_denied;
479 }
480 else if (access_permission == MapAccessPermissions::READ_ONLY)
481 {
482 set_fn = js_kv_map_set_denied;
483 delete_fn = js_kv_map_delete_denied;
484 clear_fn = js_kv_map_clear_denied;
485 }
486
487 auto has_fn_val = ctx.new_c_function(has_fn, "has", 1);
488 JS_CHECK_EXC(has_fn_val);
489 JS_CHECK_SET(view_val.set("has", std::move(has_fn_val)));
490
491 auto get_fn_val = ctx.new_c_function(get_fn, "get", 1);
492 JS_CHECK_EXC(get_fn_val);
493 JS_CHECK_SET(view_val.set("get", std::move(get_fn_val)));
494
495 auto get_size_fn_val = ctx.new_getter_c_function(size_fn, "size");
496 JS_CHECK_EXC(get_size_fn_val);
497 JS_CHECK_SET(view_val.set_getter("size", std::move(get_size_fn_val)));
498
499 auto set_fn_val = ctx.new_c_function(set_fn, "set", 2);
500 JS_CHECK_EXC(set_fn_val);
501 JS_CHECK_SET(view_val.set("set", std::move(set_fn_val)));
502
503 auto delete_fn_val = ctx.new_c_function(delete_fn, "delete", 1);
504 JS_CHECK_EXC(delete_fn_val);
505 JS_CHECK_SET(view_val.set("delete", std::move(delete_fn_val)));
506
507 auto clear_fn_val = ctx.new_c_function(clear_fn, "clear", 0);
508 JS_CHECK_EXC(clear_fn_val);
509 JS_CHECK_SET(view_val.set("clear", std::move(clear_fn_val)));
510
511 auto foreach_fn_val = ctx.new_c_function(foreach_fn, "forEach", 1);
512 JS_CHECK_EXC(foreach_fn_val);
513 JS_CHECK_SET(view_val.set("forEach", std::move(foreach_fn_val)));
514
515 auto get_version_fn_val =
516 ctx.new_c_function(get_version_fn, "getVersionOfPreviousWrite", 1);
517 JS_CHECK_EXC(get_version_fn_val);
518 JS_CHECK_SET(view_val.set(
519 "getVersionOfPreviousWrite", std::move(get_version_fn_val)));
520
521 return view_val.take();
522 }

525 static int js_kv_lookup(
526 JSContext* ctx,
527 JSPropertyDescriptor* desc,
528 JSValueConst this_val,
529 JSAtom property)
530 {
531 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
532 const auto map_name = jsctx.to_str(property).value_or("");
533 LOG_TRACE_FMT("Looking for kv map '{}'", map_name);
534
535 const auto access_permission = check_kv_map_access(jsctx.access, map_name);
536 auto handle_val = _create_kv_map_handle<_get_map_handle_current>(
537 jsctx, map_name, access_permission);
538 if (JS_IsException(handle_val))
539 {
540 return -1;
541 }
542
543 desc->flags = 0;
544 desc->value = handle_val;
545
546 return true;
