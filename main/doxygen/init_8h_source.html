22 static JSValue js_random_impl(
23 JSContext* ctx, JSValueConst this_val, int argc, JSValueConst* argv)
24 {
25 crypto::EntropyPtr entropy = crypto::get_entropy();
26
27 // Generate a random 64 bit unsigned int, and transform that to a double
28 // between 0 and 1. Note this is non-uniform, and not cryptographically
29 // sound.
30 union
31 {
32 double d;
33 uint64_t u;
34 } u;
35 try
36 {
37 u.u = entropy->random64();
38 }
39 catch (const std::exception& e)
40 {
41 return JS_ThrowInternalError(
42 ctx, "Failed to generate random number: %s", e.what());
43 }
44 // From QuickJS - set exponent to 1, and shift random bytes to
45 // fractional part, producing 1.0 <= u.d < 2
46 u.u = ((uint64_t)1023 << 52) | (u.u >> 12);
47
48 return JS_NewFloat64(ctx, u.d - 1.0);
49 }
