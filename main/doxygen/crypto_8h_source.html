17 static JSValue js_generate_aes_key(
18 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
19 {
20 if (argc != 1)
21 return JS_ThrowTypeError(
22 ctx, "Passed %d arguments, but expected 1", argc);
23
24 int32_t key_size;
25 if (JS_ToInt32(ctx, &key_size, argv[0]) < 0)
26 {
27 return ccf::js::core::constants::Exception;
28 }
29 // Supported key sizes for AES.
30 if (key_size != 128 && key_size != 192 && key_size != 256)
31 {
32 return JS_ThrowRangeError(
33 ctx, "invalid key size (not one of 128, 192, 256)");
34 }
35
36 try
37 {
38 std::vector<uint8_t> key = crypto::get_entropy()->random(key_size / 8);
39 return JS_NewArrayBufferCopy(ctx, key.data(), key.size());
40 }
41 catch (const std::exception& exc)
42 {
43 return JS_ThrowInternalError(
44 ctx, "Failed to generate AES key: %s", exc.what());
45 }

48 static JSValue js_generate_rsa_key_pair(
49 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
50 {
51 if (argc != 1 && argc != 2)
52 return JS_ThrowTypeError(
53 ctx, "Passed %d arguments, but expected 1 or 2", argc);
54
55 uint32_t key_size = 0, key_exponent = 0;
56 if (JS_ToUint32(ctx, &key_size, argv[0]) < 0)
57 {
58 return ccf::js::core::constants::Exception;
59 }
60
61 if (argc == 2 && JS_ToUint32(ctx, &key_exponent, argv[1]) < 0)
62 {
63 return ccf::js::core::constants::Exception;
64 }
65
66 std::shared_ptr<crypto::RSAKeyPair> k;
67 try
68 {
69 if (argc == 1)
70 {
71 k = crypto::make_rsa_key_pair(key_size);
72 }
73 else
74 {
75 k = crypto::make_rsa_key_pair(key_size, key_exponent);
76 }
77 }
78 catch (const std::exception& exc)
79 {
80 return JS_ThrowInternalError(
81 ctx, "Failed to generate RSA key pair: %s", exc.what());
82 }
83
84 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
85
86 try
87 {
88 crypto::Pem prv = k->private_key_pem();
89 crypto::Pem pub = k->public_key_pem();
90
91 auto r = jsctx.new_obj();
92 JS_CHECK_EXC(r);
93 auto private_key = jsctx.new_string_len((char*)prv.data(), prv.size());
94 OPENSSL_cleanse(prv.data(), prv.size());
95 JS_CHECK_EXC(private_key);
96 JS_CHECK_SET(r.set("privateKey", std::move(private_key)));
97 auto public_key = jsctx.new_string_len((char*)pub.data(), pub.size());
98 JS_CHECK_EXC(public_key);
99 JS_CHECK_SET(r.set("publicKey", std::move(public_key)));
100
101 return r.take();
102 }
103 catch (const std::exception& exc)
104 {
105 return JS_ThrowInternalError(
106 ctx, "Failed to serialise RSA key pair: %s", exc.what());
107 }
108 }

296 static JSValue js_is_valid_x509_cert_chain(
297 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
298 {
299 // first arg: chain (concatenated PEM certs, first cert = target)
300 // second arg: trusted (concatenated PEM certs)
301 if (argc != 2)
302 return JS_ThrowTypeError(
303 ctx, "Passed %d arguments, but expected 2", argc);
304
305 auto chain_js = argv[0];
306 auto trusted_js = argv[1];
307
308 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
309
310 auto chain_str = jsctx.to_str(chain_js);
311 if (!chain_str)
312 {
313 return ccf::js::core::constants::Exception;
314 }
315 auto trusted_str = jsctx.to_str(trusted_js);
316 if (!trusted_str)
317 {
318 return ccf::js::core::constants::Exception;
319 }
320
321 try
322 {
323 auto chain_vec = crypto::split_x509_cert_bundle(*chain_str);
324 auto trusted_vec = crypto::split_x509_cert_bundle(*trusted_str);
325 if (chain_vec.empty() || trusted_vec.empty())
326 {
327 throw std::runtime_error(
328 "chain/trusted arguments must contain at least one certificate");
329 }
330
331 auto& target_pem = chain_vec[0];
332 std::vector<const crypto::Pem*> chain_ptr;
333 for (auto it = chain_vec.begin() + 1; it != chain_vec.end(); it++)
334 {
335 chain_ptr.push_back(&*it);
336 }
337 std::vector<const crypto::Pem*> trusted_ptr;
338 for (auto& pem : trusted_vec)
339 {
340 trusted_ptr.push_back(&pem);
341 }
342
343 auto verifier = crypto::make_unique_verifier(target_pem);
344 if (!verifier->verify_certificate(trusted_ptr, chain_ptr))
345 {
346 throw std::runtime_error("certificate chain is invalid");
347 }
348 }
349 catch (const std::runtime_error& e)
350 {
351 LOG_DEBUG_FMT("isValidX509Chain: {}", e.what());
352 return ccf::js::core::constants::False;
353 }
354 catch (const std::logic_error& e)
355 {
356 return JS_ThrowInternalError(
357 ctx, "isValidX509Chain failed: %s", e.what());
358 }
359
360 return ccf::js::core::constants::True;
361 }

363 template <typename T>
364 static JSValue js_pem_to_jwk(
365 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
366 {
367 if (argc != 1 && argc != 2)
368 return JS_ThrowTypeError(
369 ctx, "Passed %d arguments, but expected 1 or 2", argc);
370
371 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
372
373 auto pem_str = jsctx.to_str(argv[0]);
374 if (!pem_str)
375 {
376 return ccf::js::core::constants::Exception;
377 }
378
379 std::optional<std::string> kid = std::nullopt;
380 if (argc == 2)
381 {
382 auto kid_str = jsctx.to_str(argv[1]);
383 if (!kid_str)
384 {
385 return ccf::js::core::constants::Exception;
386 }
387 kid = kid_str;
388 }
389
390 T jwk;
391 try
392 {
393 if constexpr (std::is_same_v<T, crypto::JsonWebKeyECPublic>)
394 {
395 auto pubk = crypto::make_public_key(*pem_str);
396 jwk = pubk->public_key_jwk(kid);
397 }
398 else if constexpr (std::is_same_v<T, crypto::JsonWebKeyECPrivate>)
399 {
400 auto kp = crypto::make_key_pair(*pem_str);
401 jwk = kp->private_key_jwk(kid);
402 }
403 else if constexpr (std::is_same_v<T, crypto::JsonWebKeyRSAPublic>)
404 {
405 auto pubk = crypto::make_rsa_public_key(*pem_str);
406 jwk = pubk->public_key_jwk_rsa(kid);
407 }
408 else if constexpr (std::is_same_v<T, crypto::JsonWebKeyRSAPrivate>)
409 {
410 auto kp = crypto::make_rsa_key_pair(*pem_str);
411 jwk = kp->private_key_jwk_rsa(kid);
412 }
413 else if constexpr (std::is_same_v<T, crypto::JsonWebKeyEdDSAPublic>)
414 {
415 auto pubk = crypto::make_eddsa_public_key(*pem_str);
416 jwk = pubk->public_key_jwk_eddsa(kid);
417 }
418 else if constexpr (std::is_same_v<T, crypto::JsonWebKeyEdDSAPrivate>)
419 {
420 auto kp = crypto::make_eddsa_key_pair(*pem_str);
421 jwk = kp->private_key_jwk_eddsa(kid);
422 }
423 else
424 {
425 static_assert(nonstd::dependent_false_v<T>, "Unknown type");
426 }
427 }
428 catch (const std::exception& ex)
429 {
430 return JS_ThrowInternalError(
431 ctx, "Failed to convert pem to jwk: %s", ex.what());
432 }
433
434 try
435 {
436 auto jwk_str = nlohmann::json(jwk).dump();
437 return JS_ParseJSON(ctx, jwk_str.c_str(), jwk_str.size(), "<jwk>");
438 }
439 catch (const std::exception& ex)
440 {
441 return JS_ThrowInternalError(
442 ctx, "Failed to serialise jwk: %s", ex.what());
443 }
444 }

513 static JSValue js_wrap_key(
514 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
515 {
516 if (argc != 3)
517 return JS_ThrowTypeError(
518 ctx, "Passed %d arguments, but expected 3", argc);
519
520 // API loosely modeled after
521 // https://developer.mozilla.org/en-US/docs/Web/API/SubtleCrypto/wrapKey.
522
523 size_t key_size;
524 uint8_t* key = JS_GetArrayBuffer(ctx, &key_size, argv[0]);
525 if (!key)
526 {
527 return ccf::js::core::constants::Exception;
528 }
529
530 size_t wrapping_key_size;
531 uint8_t* wrapping_key =
532 JS_GetArrayBuffer(ctx, &wrapping_key_size, argv[1]);
533 if (!wrapping_key)
534 {
535 return ccf::js::core::constants::Exception;
536 }
537
538 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
539
540 auto parameters = argv[2];
541 auto wrap_algo_name_val = jsctx.get_property(parameters, "name");
542 JS_CHECK_EXC(wrap_algo_name_val);
543
544 auto wrap_algo_name_str = jsctx.to_str(wrap_algo_name_val);
545 if (!wrap_algo_name_str)
546 {
547 return ccf::js::core::constants::Exception;
548 }
549
550 try
551 {
552 auto algo_name = *wrap_algo_name_str;
553 if (algo_name == "RSA-OAEP")
554 {
555 // key can in principle be arbitrary data (see note on maximum size
556 // in rsa_key_pair.h). wrapping_key is a public RSA key.
557
558 auto label_val = jsctx.get_property(parameters, "label");
559 JS_CHECK_EXC(label_val);
560
561 size_t label_buf_size = 0;
562 uint8_t* label_buf =
563 JS_GetArrayBuffer(ctx, &label_buf_size, label_val.val);
564
565 std::optional<std::vector<uint8_t>> label_opt = std::nullopt;
566 if (label_buf && label_buf_size > 0)
567 {
568 label_opt = {label_buf, label_buf + label_buf_size};
569 }
570
571 auto wrapped_key = crypto::ckm_rsa_pkcs_oaep_wrap(
572 crypto::Pem(wrapping_key, wrapping_key_size),
573 {key, key + key_size},
574 label_opt);
575
576 return JS_NewArrayBufferCopy(
577 ctx, wrapped_key.data(), wrapped_key.size());
578 }
579 else if (algo_name == "AES-KWP")
580 {
581 std::vector<uint8_t> privateKey(
582 wrapping_key, wrapping_key + wrapping_key_size);
583 std::vector<uint8_t> wrapped_key =
584 crypto::ckm_aes_key_wrap_pad(privateKey, {key, key + key_size});
585
586 OPENSSL_cleanse(privateKey.data(), privateKey.size());
587
588 return JS_NewArrayBufferCopy(
589 ctx, wrapped_key.data(), wrapped_key.size());
590 }
591 else if (algo_name == "RSA-OAEP-AES-KWP")
592 {
593 auto aes_key_size_value =
594 jsctx.get_property(parameters, "aesKeySize");
595 JS_CHECK_EXC(aes_key_size_value);
596
597 int32_t aes_key_size = 0;
598 if (JS_ToInt32(ctx, &aes_key_size, aes_key_size_value.val) < 0)
599 {
600 return ccf::js::core::constants::Exception;
601 }
602
603 auto label_val = jsctx.get_property(parameters, "label");
604 JS_CHECK_EXC(label_val);
605
606 size_t label_buf_size = 0;
607 uint8_t* label_buf =
608 JS_GetArrayBuffer(ctx, &label_buf_size, label_val.val);
609
610 std::optional<std::vector<uint8_t>> label_opt = std::nullopt;
611 if (label_buf && label_buf_size > 0)
612 {
613 label_opt = {label_buf, label_buf + label_buf_size};
614 }
615
616 auto wrapped_key = crypto::ckm_rsa_aes_key_wrap(
617 aes_key_size,
618 crypto::Pem(wrapping_key, wrapping_key_size),
619 {key, key + key_size},
620 label_opt);
621
622 return JS_NewArrayBufferCopy(
623 ctx, wrapped_key.data(), wrapped_key.size());
624 }
625 else
626 {
627 return JS_ThrowRangeError(
628 ctx,
629 "unsupported key wrapping algorithm, supported: RSA-OAEP, AES-KWP, "
630 "RSA-OAEP-AES-KWP");
631 }
632 }
633 catch (std::exception& ex)
634 {
635 return JS_ThrowInternalError(ctx, "Failed to wrap key: %s", ex.what());
636 }
637 catch (...)
638 {
639 return JS_ThrowRangeError(ctx, "caught unknown exception");
640 }
641 }

643 static JSValue js_unwrap_key(
644 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
645 {
646 if (argc != 3)
647 return JS_ThrowTypeError(
648 ctx, "Passed %d arguments, but expected 3", argc);
649
650 // API loosely modeled after
651 // https://developer.mozilla.org/en-US/docs/Web/API/SubtleCrypto/unwrapKey.
652
653 size_t key_size;
654 uint8_t* key = JS_GetArrayBuffer(ctx, &key_size, argv[0]);
655 if (!key)
656 {
657 return ccf::js::core::constants::Exception;
658 }
659
660 size_t unwrapping_key_size;
661 uint8_t* unwrapping_key =
662 JS_GetArrayBuffer(ctx, &unwrapping_key_size, argv[1]);
663 if (!unwrapping_key)
664 {
665 return ccf::js::core::constants::Exception;
666 }
667
668 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
669
670 auto parameters = argv[2];
671 auto wrap_algo_name_val = jsctx.get_property(parameters, "name");
672 JS_CHECK_EXC(wrap_algo_name_val);
673
674 auto wrap_algo_name_str = jsctx.to_str(wrap_algo_name_val);
675 if (!wrap_algo_name_str)
676 {
677 return ccf::js::core::constants::Exception;
678 }
679
680 try
681 {
682 auto algo_name = *wrap_algo_name_str;
683 if (algo_name == "RSA-OAEP")
684 {
685 // key can in principle be arbitrary data (see note on maximum size
686 // in rsa_key_pair.h). unwrapping_key is a private RSA key.
687
688 auto label_val = jsctx.get_property(parameters, "label");
689 JS_CHECK_EXC(label_val);
690
691 size_t label_buf_size = 0;
692 uint8_t* label_buf =
693 JS_GetArrayBuffer(ctx, &label_buf_size, label_val.val);
694
695 std::optional<std::vector<uint8_t>> label_opt = std::nullopt;
696 if (label_buf && label_buf_size > 0)
697 {
698 label_opt = {label_buf, label_buf + label_buf_size};
699 }
700
701 auto pemPrivateUnwrappingKey =
702 crypto::Pem(unwrapping_key, unwrapping_key_size);
703 auto unwrapped_key = crypto::ckm_rsa_pkcs_oaep_unwrap(
704 pemPrivateUnwrappingKey, {key, key + key_size}, label_opt);
705
706 OPENSSL_cleanse(
707 pemPrivateUnwrappingKey.data(), pemPrivateUnwrappingKey.size());
708
709 return JS_NewArrayBufferCopy(
710 ctx, unwrapped_key.data(), unwrapped_key.size());
711 }
712 else if (algo_name == "AES-KWP")
713 {
714 std::vector<uint8_t> privateKey(
715 unwrapping_key, unwrapping_key + unwrapping_key_size);
716 std::vector<uint8_t> unwrapped_key =
717 crypto::ckm_aes_key_unwrap_pad(privateKey, {key, key + key_size});
718
719 OPENSSL_cleanse(privateKey.data(), privateKey.size());
720
721 return JS_NewArrayBufferCopy(
722 ctx, unwrapped_key.data(), unwrapped_key.size());
723 }
724 else if (algo_name == "RSA-OAEP-AES-KWP")

727 jsctx.get_property(parameters, "aesKeySize");
728 JS_CHECK_EXC(aes_key_size_value);

731 if (JS_ToInt32(ctx, &aes_key_size, aes_key_size_value.val) < 0)
732 {
733 return ccf::js::core::constants::Exception;
734 }
735
736 auto label_val = jsctx.get_property(parameters, "label");
737 JS_CHECK_EXC(label_val);
738
739 size_t label_buf_size = 0;
740 uint8_t* label_buf =
741 JS_GetArrayBuffer(ctx, &label_buf_size, label_val.val);
742
743 std::optional<std::vector<uint8_t>> label_opt = std::nullopt;
744 if (label_buf && label_buf_size > 0)
745 {
746 label_opt = {label_buf, label_buf + label_buf_size};
747 }
748
749 auto privPemUnwrappingKey =
750 crypto::Pem(unwrapping_key, unwrapping_key_size);
751 auto unwrapped_key = crypto::ckm_rsa_aes_key_unwrap(
752 privPemUnwrappingKey, {key, key + key_size}, label_opt);
753
754 OPENSSL_cleanse(
755 privPemUnwrappingKey.data(), privPemUnwrappingKey.size());
756
757 return JS_NewArrayBufferCopy(
758 ctx, unwrapped_key.data(), unwrapped_key.size());
759 }
760 else
761 {
762 return JS_ThrowRangeError(
763 ctx,
764 "unsupported key unwrapping algorithm, supported: RSA-OAEP, "
765 "AES-KWP, "
766 "RSA-OAEP-AES-KWP");
767 }
768 }
769 catch (std::exception& ex)
770 {
771 return JS_ThrowInternalError(
772 ctx, "Failed to unwrap key: %s", ex.what());
773 }
774 catch (...)
775 {
776 return JS_ThrowRangeError(ctx, "caught unknown exception");
777 }

780 static JSValue js_sign(
781 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
782 {
783 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
784
785 if (argc != 3)
786 {
787 return JS_ThrowTypeError(
788 ctx, "Passed %d arguments, but expected 3", argc);
789 }
790
791 auto algorithm = argv[0];
792
793 auto algo_name_val = jsctx.get_property(algorithm, "name");
794 JS_CHECK_EXC(algo_name_val);
795
796 auto algo_hash_val = jsctx.get_property(algorithm, "hash");
797 JS_CHECK_EXC(algo_hash_val);
798
799 auto algo_name_str = jsctx.to_str(algo_name_val);
800 if (!algo_name_str)
801 {
802 return ccf::js::core::constants::Exception;
803 }
804
805 auto key_str = jsctx.to_str(argv[1]);
806 if (!key_str)
807 {
808 return ccf::js::core::constants::Exception;
809 }
810 auto key = *key_str;
811
812 size_t data_size;
813 uint8_t* data = JS_GetArrayBuffer(ctx, &data_size, argv[2]);
814 if (!data)
815 {
816 return ccf::js::core::constants::Exception;
817 }
818 std::vector<uint8_t> contents(data, data + data_size);
819
820 // Handle algorithms that don't use algo_hash here
821 if (*algo_name_str == "EdDSA")
822 {
823 try
824 {
825 crypto::Pem key_pem(key);
826 auto key_pair = crypto::make_eddsa_key_pair(key_pem);
827 auto sig = key_pair->sign(contents);
828 return JS_NewArrayBufferCopy(ctx, sig.data(), sig.size());
829 }
830 catch (const std::exception& ex)
831 {
832 return JS_ThrowInternalError(
833 ctx, "Failed to sign with EdDSA pair: %s", ex.what());
834 }
835 }
836
837 auto algo_hash_str = jsctx.to_str(algo_hash_val);
838 if (!algo_hash_str)
839 {
840 return ccf::js::core::constants::Exception;
841 }
842
843 try
844 {
845 auto algo_name = *algo_name_str;
846 auto algo_hash = *algo_hash_str;
847
848 crypto::MDType mdtype;
849 if (algo_hash == "SHA-256")
850 {
851 mdtype = crypto::MDType::SHA256;
852 }
853 else if (algo_hash == "SHA-384")
854 {
855 mdtype = crypto::MDType::SHA384;
856 }
857 else if (algo_hash == "SHA-512")
858 {
859 mdtype = crypto::MDType::SHA512;
860 }
861 else
862 {
863 return JS_ThrowRangeError(
864 ctx,
865 "Unsupported hash algorithm, supported: SHA-256, SHA-384, SHA-512");
866 }
867
868 if (algo_name == "ECDSA")
869 {
870 auto key_pair = crypto::make_key_pair(key);
871 auto sig_der = key_pair->sign(contents, mdtype);
872 auto sig =
873 crypto::ecdsa_sig_der_to_p1363(sig_der, key_pair->get_curve_id());
874 return JS_NewArrayBufferCopy(ctx, sig.data(), sig.size());
875 }
876 else if (algo_name == "RSASSA-PKCS1-v1_5")
877 {
878 auto key_pair = crypto::make_rsa_key_pair(key);
879 auto sig = key_pair->sign(contents, mdtype);
880 return JS_NewArrayBufferCopy(ctx, sig.data(), sig.size());
881 }
882 else if (algo_name == "HMAC")
883 {
884 std::vector<uint8_t> vkey(key.begin(), key.end());
885 const auto sig = crypto::hmac(mdtype, vkey, contents);
886 return JS_NewArrayBufferCopy(ctx, sig.data(), sig.size());
887 }
888 else
889 {
890 return JS_ThrowRangeError(
891 ctx,
892 "Unsupported signing algorithm, supported: RSASSA-PKCS1-v1_5, "
893 "ECDSA, EdDSA, HMAC");
894 }
895 }
896 catch (const std::exception& ex)
897 {
898 return JS_ThrowInternalError(ctx, "Failed to sign: %s", ex.what());
899 }
900 }

914 static JSValue js_verify_signature(
915 JSContext* ctx, JSValueConst, int argc, JSValueConst* argv)
916 {
917 js::core::Context& jsctx = *(js::core::Context*)JS_GetContextOpaque(ctx);
918
919 if (argc != 4)
920 {
921 return JS_ThrowTypeError(
922 ctx, "Passed %d arguments, but expected 4", argc);
923 }
924
925 // API loosely modeled after
926 // https://developer.mozilla.org/en-US/docs/Web/API/SubtleCrypto/verify.
927
928 size_t signature_size;
929 uint8_t* signature = JS_GetArrayBuffer(ctx, &signature_size, argv[2]);
930 if (!signature)
931 {
932 return ccf::js::core::constants::Exception;
933 }
934
935 size_t data_size;
936 uint8_t* data = JS_GetArrayBuffer(ctx, &data_size, argv[3]);
937 if (!data)
938 {
939 return ccf::js::core::constants::Exception;
940 }
941
942 auto algorithm = argv[0];
943
944 auto algo_name_val = jsctx.get_property(algorithm, "name");
945 JS_CHECK_EXC(algo_name_val);
946
947 auto algo_hash_val = jsctx.get_property(algorithm, "hash");
948 JS_CHECK_EXC(algo_hash_val);
949
950 auto algo_name_str = jsctx.to_str(algo_name_val);
951 if (!algo_name_str)
952 {
953 return ccf::js::core::constants::Exception;
954 }
955
956 auto key_str = jsctx.to_str(argv[1]);
957 if (!key_str)
958 {
959 return ccf::js::core::constants::Exception;
960 }
961
962 // Handle algorithms that don't use algo_hash here
963 if (*algo_name_str == "EdDSA")
964 {
965 try
966 {
967 return JS_NewBool(
968 ctx,
969 verify_eddsa_signature(
970 data, data_size, signature, signature_size, *key_str));
971 }
972 catch (const std::exception& ex)
973 {
974 return JS_ThrowRangeError(
975 ctx, "Failed to verify EdDSA signature: %s", ex.what());
976 }
977 }
978
979 auto algo_hash_str = jsctx.to_str(algo_hash_val);
980 if (!algo_hash_str)
981 {
982 return ccf::js::core::constants::Exception;
983 }
984
985 try
986 {
987 auto algo_name = *algo_name_str;
988 auto algo_hash = *algo_hash_str;
989 auto key = *key_str;
990
991 crypto::MDType mdtype;
992 if (algo_hash == "SHA-256")
993 {
994 mdtype = crypto::MDType::SHA256;
995 }
996 else
997 {
998 return JS_ThrowRangeError(
999 ctx, "Unsupported hash algorithm, supported: SHA-256");
1000 }
1001
1002 if (algo_name != "RSASSA-PKCS1-v1_5" && algo_name != "ECDSA")
1003 {
1004 return JS_ThrowRangeError(
1005 ctx,
1006 "Unsupported signing algorithm, supported: RSASSA-PKCS1-v1_5, "
1007 "ECDSA, EdDSA");
1008 }
1009
1010 std::vector<uint8_t> sig(signature, signature + signature_size);
1011
1012 if (algo_name == "ECDSA")
1013 {
1014 sig = crypto::ecdsa_sig_p1363_to_der(sig);
1015 }
1016
1017 auto is_cert = key.starts_with("-----BEGIN CERTIFICATE");
1018
1019 bool valid = false;
1020
1021 if (is_cert)
1022 {
1023 auto verifier = crypto::make_unique_verifier(key);
1024 valid =
1025 verifier->verify(data, data_size, sig.data(), sig.size(), mdtype);
1026 }
1027 else
1028 {
1029 auto public_key = crypto::make_public_key(key);
1030 valid =
1031 public_key->verify(data, data_size, sig.data(), sig.size(), mdtype);
1032 }
1033
1034 return JS_NewBool(ctx, valid);
1035 }
1036 catch (const std::exception& ex)
1037 {
1038 return JS_ThrowInternalError(
1039 ctx, "Failed to verify signature: %s", ex.what());
1040 }
1041 }
